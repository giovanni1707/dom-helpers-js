Components Module

Build reusable, reactive components with lifecycle hooks

Installation

<script src="https://cdn.jsdelivr.net/npm/@giovanni1707/dom-helpers@1.0.0/dist/dom-helpers-components.min.js"></script>

Creating Components

Components.register('UserCard', `
    <div class="user-card">
        <img id="avatar" />
        <h3 id="name"></h3>
        <p id="email"></p>
    </div>
    
    <style>
        .user-card {
            padding: 20px;
            border: 1px solid #ddd;
            border-radius: 8px;
        }
    </style>
    
    <script>
        Elements.update({
            "avatar.src": data.avatar,
            "name.textContent": data.name,
            "email.textContent": data.email
        });
        
        onMounted(() => {
            console.log('Component mounted!');
        });
    </script>
`);

Using Components

// Render component
Components.render('UserCard', '#container', {
    name: 'John Doe',
    email: 'john@example.com',
    avatar: 'avatar.jpg'
});

// Or use HTML tags
<UserCard 
    name="John Doe"
    email="john@example.com"
    avatar="avatar.jpg">
</UserCard>

Lifecycle Hooks

Mounting

onBeforeMount(() => {
    // Before component mounts
});

onMounted(() => {
    // After component mounts
});

Cleanup

onBeforeDestroy(() => {
    // Cleanup before destroy
    clearInterval(myInterval);
});
