Animation Module

Create smooth animations and transitions with ease

Installation

<script src="https://cdn.jsdelivr.net/npm/@giovanni1707/dom-helpers@2.0.0/dist/dom-helpers-animation.min.js"></script>

Animation Methods

Fade Effects

Animate.fadeIn(Elements.modal, 300);
Animate.fadeOut(Elements.modal, 300);
Animate.fadeToggle(Elements.modal, 300);

Slide Effects

Animate.slideDown(Elements.menu, 400);
Animate.slideUp(Elements.menu, 400);
Animate.slideToggle(Elements.menu, 400);

Custom Animations

Animate.custom(Elements.box, {
    from: { opacity: 0, transform: 'scale(0.5)' },
    to: { opacity: 1, transform: 'scale(1)' },
    duration: 500,
    easing: 'ease-out'
});

Chaining

Animate.fadeIn(Elements.alert, 300)
    .then(() => Animate.fadeOut(Elements.alert, 300));

Complete Example

// Animated modal
Elements.openModal.addEventListener('click', async () => {
    Elements.modalOverlay.style.display = 'block';
    await Animate.fadeIn(Elements.modalOverlay, 200);
    await Animate.slideDown(Elements.modalContent, 300);
});

Elements.closeModal.addEventListener('click', async () => {
    await Animate.slideUp(Elements.modalContent, 300);
    await Animate.fadeOut(Elements.modalOverlay, 200);
    Elements.modalOverlay.style.display = 'none';
});
