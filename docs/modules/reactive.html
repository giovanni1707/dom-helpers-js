Reactive Module

Reactive state management with automatic UI updates

Installation

<script src="https://cdn.jsdelivr.net/npm/@giovanni1707/dom-helpers@2.0.0/dist/dom-helpers-reactive.min.js"></script>

Create Reactive State

// Create reactive state
const state = State.create({
    count: 0,
    user: { name: 'John', age: 30 }
});

// Automatic UI updates
Elements.bind({
    counter: () => state.count,
    userName: () => state.user.name
});

// Update state - UI updates automatically
state.count++;
state.user.name = 'Jane';

Computed Properties

const state = State.create({
    firstName: 'John',
    lastName: 'Doe',
    age: 30
});

// Computed property
state.computed('fullName', () => {
    return `${state.firstName} ${state.lastName}`;
});

// Use in bindings
Elements.bind({
    fullName: () => state.fullName
});

Watch Changes

// Watch for changes
state.watch('count', (newValue, oldValue) => {
    console.log(`Count changed from ${oldValue} to ${newValue}`);
});
