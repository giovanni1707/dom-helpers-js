Quick Start

// Import the library
<script src="dom-helpers.bundle.js"></script>
<script src="dom-helpers.reactive.bundle.js"></script>
<script src="conditions.js"></script>

// Basic usage
const status = 'success';

Conditions.apply(status, {
  'success': { 
    textContent: '‚úì Operation successful!',
    style: { color: 'green' }
  },
  'error': { 
    textContent: '‚úó Operation failed',
    style: { color: 'red' }
  }
}, '#status-element');

// With reactive state
const state = ReactiveUtils.state({ count: 0 });

Conditions.whenState(
  () => state.count,
  {
    '0': { textContent: 'Zero' },
    '>0': { textContent: 'Positive' },
    '<0': { textContent: 'Negative' }
  },
  '#counter'
);
